import java.util.Scanner;

public class quiz {

    public static void main(String[] args) {
        String[] perguntas = {
                "Qual é a capital da França?",
                "Quem escreveu 'Dom Casmurro'?",
                "Qual é a montanha mais alta do mundo?",
                "Em que ano o homem chegou à Lua?",
                "Qual é a fórmula da água?",
                "Quem pintou a Mona Lisa?",
                "Qual é o maior planeta do sistema solar?",
                "O que é a fotossíntese?",
                "Qual é a língua mais falada do mundo?",
                "Qual é a moeda do Japão?",
                "Quem descobriu o Brasil?",
                "Qual é a fórmula da gravidade?",
                "Quem é conhecido como o pai da medicina?",
                "Qual é o continente mais populoso?",
                "Qual é a capital do Brasil?"
        };

        String[][] alternativas = {
                {"A) Berlim", "B) Paris", "C) Lisboa", "D) Madrid", "E) Roma"},
                {"A) Machado de Assis", "B) José de Alencar", "C) Clarice Lispector", "D) Jorge Amado", "E) Graciliano Ramos"},
                {"A) K2", "B) Everest", "C) Kilimanjaro", "D) Mont Blanc", "E) Aconcágua"},
                {"A) 1965", "B) 1969", "C) 1971", "D) 1959", "E) 1975"},
                {"A) H2O", "B) CO2", "C) O2", "D) H2O2", "E) CH4"},
                {"A) Van Gogh", "B) Picasso", "C) Da Vinci", "D) Rembrandt", "E) Matisse"},
                {"A) Terra", "B) Júpiter", "C) Marte", "D) Saturno", "E) Vênus"},
                {"A) Processo de absorção de água", "B) Produção de energia por plantas", "C) Respiração de animais", "D) Filtração do solo", "E) Ciclo da água"},
                {"A) Mandarim", "B) Espanhol", "C) Inglês", "D) Hindi", "E) Árabe"},
                {"A) Dólar", "B) Euro", "C) Iene", "D) Libra", "E) Peso"},
                {"A) Pedro Álvares Cabral", "B) Vasco da Gama", "C) Cristóvão Colombo", "D) Américo Vespúcio", "E) Fernão de Magalhães"},
                {"A) F = ma", "B) E = mc²", "C) P = mv", "D) V = IR", "E) A = l × w"},
                {"A) Hipócrates", "B) Galeno", "C) Avicena", "D) Paracelso", "E) Pasteur"},
                {"A) Ásia", "B) África", "C) Europa", "D) América", "E) Oceania"},
                {"A) Brasília", "B) São Paulo", "C) Rio de Janeiro", "D) Salvador", "E) Belo Horizonte"}
        };

        char[] respostasCorretas = {'B', 'A', 'B', 'B', 'A', 'C', 'B', 'B', 'A', 'C', 'A', 'A', 'A', 'A', 'A'};

        Scanner scanner = new Scanner(System.in);
        int acertos = 0;
        StringBuilder resultado = new StringBuilder();

        for (int i = 0; i < perguntas.length; i++) {
            System.out.println((i + 1) + ". " + perguntas[i]);
            for (String alternativa : alternativas[i]) {
                System.out.println(alternativa);
            }
            System.out.print("Digite sua resposta (A/B/C/D/E): ");
            char respostaUsuario = scanner.nextLine().toUpperCase().charAt(0);

            if (respostaUsuario == respostasCorretas[i]) {
                acertos++;
                resultado.append("Pergunta ").append(i + 1).append(": Correta!\n");
            } else {
                resultado.append("Pergunta ").append(i + 1).append(": Errada! Resposta correta: ").append(respostasCorretas[i]).append("\n");
            }
        }

        System.out.println("\nVocê acertou " + acertos + " de " + perguntas.length + " perguntas.");
        System.out.println(resultado.toString());

        scanner.close();
    }
}
